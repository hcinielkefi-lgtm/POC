3. Kamelet referentiel-loader (lecture des règles en Ingestion)

Rôle : appeler l'API RéférentielFlux (ou DB via un service) pour récupérer les paramètres et les injecter dans les headers/propriétés du message, avant toute autre étape.

apiVersion: camel.apache.org/v1
kind: Kamelet
metadata:
  name: referentiel-loader
spec:
  definition:
    title: Referentiel Loader
    type: object
    properties:
      referentielUrl:
        type: string
        default: "http://referentiel-flux.svc/api/flux"
  flow:
    from:
      uri: kamelet:source
      steps:
        - setHeader:
            name: fluxType
            simple: "${header.CamelFileNameOnly}"
        - toD: "{{referentielUrl}}?type=${header.fluxType}&channel=${header.channel}"
        - unmarshal:
            json: {}
        - setHeader: { name: rail,         simple: "${body[rail]}" }
        - setHeader: { name: mode,         simple: "${body[mode]}" }
        - setProperty: { name: needSplit,  simple: "${body[needSplit]}" }
        - setProperty: { name: chunkSize,  simple: "${body[chunkSize]}" }
        - setProperty: { name: splitExpr,  simple: "${body[splitExpr]}" }
        - setProperty: { name: cmdMapping, simple: "${body[cmdMapping]}" }
        - to: "kamelet:sink"

11. Annexes — DDL SQL & API du référentiel

CREATE TABLE FLUX_RULES (
  id SERIAL PRIMARY KEY,
  type VARCHAR(64) NOT NULL,
  channel VARCHAR(16) NOT NULL,
  source_format VARCHAR(16) NOT NULL,
  cmd_mapping VARCHAR(128) NOT NULL,
  need_split BOOLEAN NOT NULL DEFAULT FALSE,
  split_expr VARCHAR(256),
  chunk_size INT,
  rail VARCHAR(16) NOT NULL,      -- URGENT | STANDARD
  mode VARCHAR(16) NOT NULL,      -- UNIT | BATCH
  route_target VARCHAR(64) NOT NULL,
  active BOOLEAN NOT NULL DEFAULT TRUE,
  valid_from TIMESTAMP
);

CREATE TABLE OUTPUT_RULES (
  id SERIAL PRIMARY KEY,
  flux_type VARCHAR(64) NOT NULL,
  output_name VARCHAR(64) NOT NULL,
  output_format VARCHAR(32) NOT NULL,
  endpoint_type VARCHAR(16) NOT NULL,
  endpoint_uri VARCHAR(256) NOT NULL,
  active BOOLEAN NOT NULL DEFAULT TRUE
);
